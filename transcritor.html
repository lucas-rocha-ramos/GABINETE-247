export const config = { // Utilizar o Edge Runtime é essencial na Vercel para uploads e para // evitar o limite de tempo estrito das funções Serverless gratuitas. runtime: 'edge', }; export default async function handler(req) { // Bloqueia qualquer requisição que não seja POST if (req.method !== 'POST') { return new Response(JSON.stringify({ error: 'Método não permitido' }), { status: 405, headers: { 'Content-Type': 'application/json' } }); } try { // Recebe o arquivo enviado pelo nosso frontend (index.html) const formData = await req.formData(); const file = formData.get('file'); if (!file) { return new Response(JSON.stringify({ error: 'Nenhum arquivo enviado.' }), { status: 400, headers: { 'Content-Type': 'application/json' } }); } // Monta o pacote de dados para enviar à OpenAI const openaiFormData = new FormData(); openaiFormData.append('file', file); openaiFormData.append('model', 'whisper-1'); // O formato 'verbose_json' é o que garante que a OpenAI devolva os // tempos exatos (timestamps) para cada fala separada. openaiFormData.append('response_format', 'verbose_json'); // Envia o arquivo de áudio/vídeo para a API do Whisper const openaiResponse = await fetch('https://api.openai.com/v1/audio/transcriptions', { method: 'POST', headers: { 'Authorization': `Bearer ${process.env.OPENAI_API_KEY}` }, body: openaiFormData }); const data = await openaiResponse.json(); // Se a OpenAI retornar algum erro (ex: chave de API inválida, arquivo ruim) if (!openaiResponse.ok) { throw new Error(data.error?.message || 'Erro desconhecido ao processar na OpenAI'); } // Devolve os dados da transcrição com sucesso para o frontend return new Response(JSON.stringify(data), { status: 200, headers: { 'Content-Type': 'application/json' } }); } catch (error) { // Captura e devolve qualquer erro ocorrido no processo return new Response(JSON.stringify({ error: error.message }), { status: 500, headers: { 'Content-Type': 'application/json' } }); } }
